source code

                    const int analogOutPin = 3; //A0; --> not working
                    
                    void setup() {
                      Serial.begin(9600);
                    }
                    
                    void loop() {
                      for(int i=0; i<=255; i++){
                        analogWrite(analogOutPin, i);
                        delay(10);
                      }
                      for(int i=255; i>=0; i--){
                        analogWrite(analogOutPin, i);
                        delay(10);
                      }
                    }

NOTE:

  • when using the analogWrite() function there is no need to call the pinMode() function
    Calling directly analogWrite(pinNumber, value) is enough
  • the analog output can be done by the PWM pins alone (pins marked with ~)
