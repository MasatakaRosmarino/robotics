source code

                    const int lightIn = A0;
                    const int lightOut = 3;
                    int mappedInput;
                    int mappedOutput;
                    
                    void setup() {
                      Serial.begin(9600);
                    }
                    
                    void loop() {
                      mappedInput = analogRead(lightIn);
                    
                      mappedOutput = map(mappedInput, 0, 1023, 0, 255);
                    
                      Serial.println(analogRead(lightIn));
                      
                      analogWrite(lightOut, mappedOutput);
                      delay(300);
                    }
